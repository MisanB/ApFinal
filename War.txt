import java.util.ArrayList;

public class Deck {
    
    static final int spadesIndex = 0;
    static final int diamondsIndex = 1;
    static final int heartsIndex = 2;
    static final int clubsIndex = 3;
    
    ArrayList<card> DeckOfCards = new ArrayList<>();
    
    public Deck()
    {
        DeckOfCards.addAll(initSuit(spadesIndex));
        DeckOfCards.addAll(initSuit(diamondsIndex));
        DeckOfCards.addAll(initSuit(heartsIndex));
        DeckOfCards.addAll(initSuit(clubsIndex));
    }
    public Deck(boolean off)
    {
        return;
    }
    
    public card pickCard()
    {
        return DeckOfCards.get( (int)(Math.random() * DeckOfCards.size()));
    }
    public int pickIndex()
    {
        return (int)(Math.random() * DeckOfCards.size());
    }
    
    private ArrayList<card> initSuit(int Suit)
    {
        ArrayList<card> Cards = new ArrayList<>();
        
        for (int index =  2 ; index < 15 ; index++)
        {
            card newCard = new card(Suit, index);
            Cards.add(newCard);
        }
        
        return Cards;
    }
    
    public card get(int index)
    {
        return DeckOfCards.get(index);
    }
    public void add(card obj)
    {
        DeckOfCards.add(obj);
    }
    public void add(int index, card obj)
    {
        DeckOfCards.add(index, obj);
    }
    public void remove(int index)
    {
        DeckOfCards.remove(index);
    }
    public int size()
    {
        return DeckOfCards.size();
    }
    public void clear()
    {
        DeckOfCards.clear();
    }
    
    static String getSuit(int index)
    {
        if (spadesIndex == index)
        {
            return "Spades";
        }
        if (diamondsIndex == index)
        {
            return "Diamonds";
        }
        if (heartsIndex == index)
        {
            return "Hearts";
        }
        else
        {
            return "Clubs";
        }
    }
}

import java.util.*;
import java.util.Scanner; 

public class MyProgram
{
    static Scanner keyboard = new Scanner(System.in);
    
    static Deck CardDeck = new Deck();
    
    static Deck PlayerDeck = new Deck(false);
    static Deck AiDeck = new Deck(false);
    
    static int PlayerMoney = 1000;
    
    static int amountPerDeck = 26;
    
    public static void main(String[] args)
    {
        GameSet();
        
        System.out.println(" Welcome to the warcino ! your starting balence is $" + PlayerMoney);
        
        System.out.println(" Type \"bet\" to bet anyTime ! ");
        
        while( (PlayerDeck.size() > 0 && AiDeck.size() > 0) )
        {
            sim(true);
            
            System.out.println("");
            
            System.out.println(" Your chances of winning " + chances(PlayerDeck.size()) + " %");
            
            int CurrentMoneyline = Moneyline();
            
            System.out.println(" Current Moneyline " + CurrentMoneyline + " ");
            
            if ("bet".equals(keyboard.nextLine()) && PlayerMoney > 0)
            {
                System.out.print(" what are your stakes (balance: " + PlayerMoney + ")? ");
                
                int stakes = Math.min(PlayerMoney, numInput());
                
                System.out.println(" whole game or set ? w/s ");
                
                boolean win;
                
                if (keyboard.next().contains("w"))
                {
                    while((PlayerDeck.size() > 0 && AiDeck.size() > 0))
                    {
                        sim(false);
                    }
                    win = PlayerDeck.size() > 0;
                }
                else
                {
                    win = sim(true);
                }
                
                if (win)
                {
                    int winnings = PayOut(stakes, CurrentMoneyline);
                    PlayerMoney += winnings;
                    
                    System.out.println("\n Wow you won $" + winnings);
                }
                else
                {
                    PlayerMoney -= stakes;
                    System.out.println("\n Sorry pal, keep spending Im sure luck will find you");
                }
            }
            
            keyboard.nextLine();
            
            if (PlayerDeck.size() == 0 || AiDeck.size() == 0)
            {
                System.out.println("");
                System.out.print("would you like to play again ? y/n ");
                
                if ("y".equals(keyboard.nextLine())) 
                {
                    GameSet();
                }
            }
            
            System.out.println("");
            
            
        }
        
    }
    
    static public void GameSet()
    {
        Collections.shuffle(CardDeck.DeckOfCards);
        
        System.out.print(" How many cards would you like to play with in each deck? ");
        
        amountPerDeck = Math.min(numInput(), 26);
        
        System.out.println("");
        
        AiDeck.clear();
        PlayerDeck.clear();
        
        for (int index = 0 ; index < amountPerDeck * 2 ; index++)
        {
            if (index < amountPerDeck)
            {
                PlayerDeck.add(CardDeck.get(index));
            }
            else
            {
                AiDeck.add(CardDeck.get(index));
            }
        }
    }
    
    static public int Moneyline()
    {
        double chance = (chances(PlayerDeck.size()) / 100.0);
        
        if (chance > .5)
        {
            return (int)((chance / (1 - chance)) * -100);
        }
        else
        {
            return (int)(((1 - chance) / chance) * 100);
        }
    }
    static public int PayOut(int stakes, int odds)
    {
        if (odds < 0)
        {
            return (int)((100 / odds) * stakes);
        }
        else
        {
            return (int)((odds / 100) * stakes);
        }
    }
    static public double chances(int size)
    {
        return (size / (amountPerDeck * 2.0)) * 100;
    }
    
    static public boolean sim(boolean printing)
    {
        int AiIndex = AiDeck.pickIndex();
        int PlayerIndex = PlayerDeck.pickIndex();
        
        card PlayerCard = PlayerDeck.get(PlayerIndex);
        card AiCard = AiDeck.get(AiIndex);
        
        if (printing) System.out.print(PlayerCard + " vs. " + AiCard);
        
        if (PlayerDeck.get(PlayerIndex).is_greater(AiDeck.get(AiIndex)))
        {
            PlayerDeck.add(AiDeck.get(AiIndex));
            AiDeck.remove(AiIndex);
            
            if (printing) System.out.println(" Player Wins! ");
            
            return true;
        }
        else
        {
            AiDeck.add(PlayerDeck.get(PlayerIndex));
            PlayerDeck.remove(PlayerIndex);
            
           if (printing)  System.out.println(" Ai Wins! ");
           
           return false;
        }
    }
    
    public static int numInput()
    {
        return keyboard.nextInt();
    }
    
}

public class card {
    
    private int Suit; //index
    private int Rank;
    
    public card(int Suit, int Rank)
    {
        this.Suit = Suit;
        this.Rank = Rank;
    }
    
    public int getRank()
    {
        return Rank;
    }
    public int getSuit()
    {
        return Suit;
    }
    
    public String toString()
    {
        if (Rank < 11)
        {
            return Rank + " of " + Deck.getSuit(Suit);
        }
        else if (Rank == 11)
        {
            return "jack of " + Deck.getSuit(Suit);
        }
        else if (Rank == 12)
        {
            return "Queen of " + Deck.getSuit(Suit);
        }
        else if (Rank == 13)
        {
            return "King of " + Deck.getSuit(Suit);
        }
        else // if (Rank == 14)
        {
            return "Ace of " + Deck.getSuit(Suit);
        }
    }
    
    public boolean is_greater(card Other)
    {
        if (Rank != Other.getRank())
        {
            return Rank > Other.getRank();
        }
        else
        {
            return Suit < Other.getSuit();
        }
    }
}
