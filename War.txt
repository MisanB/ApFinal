import java.util.*;
import java.util.Scanner; 

public class MyProgram
{
    static Scanner keyboard = new Scanner(System.in);

    public static void printMenu()

    public static void playGoFish()

        boolean found = false;
        
        for (card cards : AiDeck.DeckOfCards)
        {
            if (cards.getRank() == rank) found = true;
        }
        
        if (found)
        {
            for (int i = PlayerDeck.size() - 1 ; i > 0 ; i--)
            {
                card cards = PlayerDeck.get(i);
                
                if (cards.getRank() == rank)
                {
                    System.out.println("AI took your " + cards);
                    AiDeck.takeCard(PlayerDeck, i);
                }
            }
        }
        else
        {
            AiDeck.dealCards(CardDeck, 1);
        }
        
    }

    static public void FishGameSet()
    {
        CardDeck.initDeck();
        
        Collections.shuffle(CardDeck.DeckOfCards);
        
        amountPerDeck = 7;
        
        System.out.print(" How many cards would you like to play with in the pile? ");
        
        amountPerDeck = Math.min(numInput(), 52 - 14);
        
        AiDeck.clear();
        PlayerDeck.clear();
        
        AiDeck.dealCards(CardDeck, amountPerDeck);
        PlayerDeck.dealCards(CardDeck, amountPerDeck);
    }
    
    public static void playWar()
    {
        System.out.println(" Welcome to the war table ! ");
        
        WarGameSet();
        
        System.out.println(" Type \"bet\" to bet anyTime ! ");
        
        while( (PlayerDeck.size() > 0 && AiDeck.size() > 0) )
        {
            sim(true);
            
            System.out.println("");
            
            System.out.println(" Your chances of winning " + chances(PlayerDeck.size()) + " %");
            
            int CurrentMoneyline = Moneyline();
            
            System.out.println(" Current Moneyline " + CurrentMoneyline + " ");
            
            if (keyboard.nextLine().contains("bet") && PlayerMoney > 0)
            {
                System.out.print(" what are your stakes (balance: " + PlayerMoney + ")? ");
                
                int stakes = Math.min(PlayerMoney, numInput());
                
                System.out.println(" whole game or set ? w/s ");
                
                boolean win;
                
                if (keyboard.next().contains("w"))
                {
                    while((PlayerDeck.size() > 0 && AiDeck.size() > 0))
                    {
                        sim(false);
                    }
                    win = PlayerDeck.size() > 0;
                }
                else
                {
                    win = sim(true);
                }
                
                if (win)
                {
                    int winnings = PayOut(stakes, CurrentMoneyline);
                    PlayerMoney += winnings;
                    
                    System.out.println("\n Wow you won $" + winnings);
                }
                else
                {
                    PlayerMoney -= stakes;
                    System.out.println("\n Sorry pal, keep spending Im sure luck will find you");
                }
            }
            
            keyboard.nextLine();
            
            if (PlayerDeck.size() == 0 || AiDeck.size() == 0)
            {
                System.out.println("");
                System.out.print("would you like to play again ? y/n ");
                
                if ("y".equals(keyboard.nextLine())) 
                {
                    WarGameSet();
                }
            }
            
            System.out.println("");
            
            
        }
        
    }
    
    static public void WarGameSet()
    {
        Collections.shuffle(CardDeck.DeckOfCards);
        
        System.out.print(" How many cards would you like to play with in each deck? ");
        
        amountPerDeck = Math.min(numInput(), 26);
        
        System.out.println("");
        
        AiDeck.clear();
        PlayerDeck.clear();
        
        for (int index = 0 ; index < amountPerDeck * 2 ; index++)
        {
            if (index < amountPerDeck)
            {
                PlayerDeck.add(CardDeck.get(index));
            }
            else
            {
                AiDeck.add(CardDeck.get(index));
            }
        }
    }

    static public boolean BeggerPenalty(String name, hand HandUsed, String otherName, hand otherHand, card initCard)
    {
        int pAmount;
        
        if (initCard.getRank() == 1)
        {
            pAmount = 4;
        }
        else
        {
            pAmount = initCard.getRank() - 10;
        }
        
        for (int i = 0 ; i < pAmount ; i++)
        {
            card AiCard = HandUsed.takeOutCard(0);
            
            if (AiCard == null)
            {
                return name.equals("Player");
            }
            
            CardDeck.add(AiCard);
            
            System.out.println(name + " deposited " + AiCard );
            
            if (AiCard.getRank() == 1 || AiCard.getRank() >= 11) 
            {
                return BeggerPenalty(otherName, otherHand, name, HandUsed, AiCard);
            }
        }
        
        return otherName.equals("Player");
    }
    
    static public Object[] BeggerDraw(String name, hand HandUsed)
    {
        card AiCard = HandUsed.takeOutCard(0);
        
        CardDeck.add(AiCard);
        
        if (AiCard == null)
        {
            Object[] returner = {true};
            return returner;
        }
        
        if (AiCard.getRank() == 1 || AiCard.getRank() > 11) 
        {
            System.out.println(name + " drew a penalty from a " + AiCard);
            
            Object[] returner = {false, AiCard};
            return returner;
        }
        else
        {
            System.out.println(name + " drew " + AiCard);
            
            Object[] returner = {true};
            return returner;
        }
    }
    
    static public void BeggerGameSet()
    {
        CardDeck.initDeck();
        
        Collections.shuffle(CardDeck.DeckOfCards);
        
        amountPerDeck = 26;
        
        AiDeck.dealCards(CardDeck, amountPerDeck);
        PlayerDeck.dealCards(CardDeck, amountPerDeck);
    }
    
    static public int Moneyline()
    {
        double chance = (chances(PlayerDeck.size()) / 100.0);
        
        if (chance > .5)
        {
            return (int)((chance / (1 - chance)) * -100);
        }
        else
        {
            return (int)(((1 - chance) / chance) * 100);
        }
    }
    static public int PayOut(int stakes, int odds)
    {
        if (odds < 0)
        {
            return (int)((100.0 / odds) * -stakes);
        }
        else
        {
            return (int)((odds / 100.0) * stakes);
        }
    }
    static public double chances(int size)
    {
        return (size / (amountPerDeck * 2.0)) * 100;
    }
    
    static public boolean sim(boolean printing)
    {
        int AiIndex = AiDeck.pickIndex();
        int PlayerIndex = PlayerDeck.pickIndex();
        
        card PlayerCard = PlayerDeck.get(PlayerIndex);
        card AiCard = AiDeck.get(AiIndex);
        
        if (printing) System.out.print(PlayerCard + " vs. " + AiCard);
        
        if (PlayerDeck.get(PlayerIndex).is_greater(AiDeck.get(AiIndex)))
        {
            PlayerDeck.add(AiDeck.get(AiIndex));
            AiDeck.remove(AiIndex);
            
            if (printing) System.out.println(" Player Wins! ");
            
            return true;
        }
        else
        {
            AiDeck.add(PlayerDeck.get(PlayerIndex));
            PlayerDeck.remove(PlayerIndex);
            
           if (printing)  System.out.println(" Ai Wins! ");
           
           return false;
        }
    }
    
    public static int numInput()
    {
        String output;
        Integer value = null;
        do
        {
            output = keyboard.next();
            
            try
            {
                value = Integer.parseInt(output);
            }
            catch (NumberFormatException error)
            {
                System.out.println("Please input only whole numbers");
            }
        }
        while (value == null);
        
        return value;
    }
